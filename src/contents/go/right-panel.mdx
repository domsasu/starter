## Hello World

```go
package main

import "fmt"

func main() {
    fmt.Println("Hello, Go!")
}
```

## Variable Declarations

```go

var global = "I'm global"

    // Different ways to declare

    var b = 20
    c := 30
    
    // Multiple variables
    var x, y int = 1, 2
    var (
        name   string = "Go"
        version float64 = 1.21
    )

```

## Type Conversions

```go
var i int = 42
var f float64 = float64(i)
var u uint = uint(f)

// Or more simply
i := 42
f := float64(i)
u := uint(f)
```

## Arrays and Slices

```go
// Arrays (fixed size)
var a [5]int
a[0] = 10
a[1] = 20

// Slices (dynamic)
var s []int
s = append(s, 1)
s = append(s, 2, 3, 4)

// Make slice
slice := make([]int, 5)    // len=5
slice2 := make([]int, 0, 5) // len=0, cap=5

// Slice literal
nums := []int{1, 2, 3, 4, 5}
```

## Maps

```go
// Declare
var m map[string]int
m = make(map[string]int)
m["answer"] = 42

// Map literal
scores := map[string]int{
    "Alice": 95,
    "Bob":   87,
}

// Check existence
value, exists := scores["Charlie"]
if exists {
    fmt.Println(value)
}

// Delete
delete(scores, "Bob")
```

## Function Examples

```go
// Simple function
func greet(name string) string {
    return "Hello, " + name
}

// Multiple returns
func divmod(a, b int) (int, int) {
    return a / b, a % b
}

// Variadic function
func sum(nums ...int) int {
    total := 0
    for _, n := range nums {
        total += n
    }
    return total
}

// Function as value
var fn func(int) int
fn = func(x int) int {
    return x * x
}
```

## Struct Example

```go
type Person struct {
    Name string
    Age  int
}

// Create instances
p1 := Person{"Alice", 30}
p2 := Person{Name: "Bob", Age: 25}
p3 := new(Person)
p3.Name = "Charlie"
p3.Age = 35
```

## Quick Commands

```bash

go run main.go

# Build executable
go build main.go

# Format code
go fmt main.go

# Get dependencies
go get github.com/pkg/errors

# Run tests
go test ./...

# Install a tool
go install github.com/tool/cmd@latest
```
